option(ENABLE_RANDOM "Enable the Random Value Plugin" ON)
option(ENABLE_SECOND_TIMER "Enable the Second Timer Plugin" ON)
option(ENABLE_TIMER "Enable the Timer Plugin" ON)

if(ENABLE_RANDOM)

set(SRC_RANDOM
    random/random.cpp
)

add_library(
    lc3_random
    SHARED
    ${SRC_RANDOM}
)

target_link_libraries(
    lc3_random
    ${lc3_LIBRARIES}
)

install(TARGETS lc3_random LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR} RUNTIME DESTINATION ${CMAKE_INSTALL_LIBDIR})

endif(ENABLE_RANDOM)

endif(ENABLE_SECOND_TIMER)
